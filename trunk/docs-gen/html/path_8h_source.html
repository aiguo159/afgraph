00085       return;
00086     }
00087     int ntmp = ndest, ncount = 0;
00088     while( ntmp != nsource && ntmp >= 0 && ntmp < n && ncount++ < n ) {
00089       push_front( ntmp );

00091     }
00092     // while can be in an infinite loop if vpred is deliberately constructed.
00093     // use a counter to break that: any path can never be more than n hops

00097       clear();
00098     }
00099     // nsource can't be reversely reached from ndest, set path empty
00100 
00101     push_front( nsource );
