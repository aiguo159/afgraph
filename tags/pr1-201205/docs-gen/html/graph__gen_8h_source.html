00077     return;
00078   }

00081 
00082   int i, j;
00083   // insert all vertices
00084   for( i = 0; i < n; ++i )
00085     for( j = 0; j < m; ++j ) {
00086       grf.insert_v( xyVertex<int>( i, j ) );

00090   for( j = 0; j < m - 1; ++j ) {
00091     grf.insert_2e( xyVertex<int>( i, j ), xyVertex<int>( i, j + 1 ), 1 );
00092   }
00093   j = 0;
00094   for( i = 1; i < n; ++i ) {
00095     grf.insert_2e( xyVertex<int>( i - 1, j ), xyVertex<int>( i, j ), 1 );

00097 
00098   for( i = 1; i < n; ++i )
00099     for( j = 1; j < m; ++j ) {
00100       grf.insert_2e( xyVertex<int>( i - 1, j ), xyVertex<int>( i, j ), 1 );
00101       grf.insert_2e( xyVertex<int>( i, j ), xyVertex<int>( i, j - 1 ), 1 );

00123 int bmw_graph_gen( T_xydGraph &dgrf,
00124            int ngrid, int n, int ntry = 10,
00125            double lamda = 0.3, double rou = 0.3 )
00126 {
00127   std::vector<std::pair<int, int> > vpos( n );
00128   double dtmp, ddist;
00129   double dmaxdist = 1.414 * ngrid;
00130   double dInfty = n * dmaxdist;
00131   int nlinks = 0;
00132   bool bconnected;
00133 
00134   while( ntry > 0 ) {
00135     std::vector<bool> vmarks( ngrid * ngrid, false ) ;
00136     int ncount = 0, i, j;
00137     nlinks = 0;
00138     // randomly choose a number of grid points as nodes
00139     while( ncount < n ) {
00140       i = ( int )afl::rand::gen( ngrid );
00141       j = ( int )afl::rand::gen( ngrid );
00142       if( !vmarks[i*ngrid+j] ) {
00143         vmarks[i *ngrid+j] = true;
00144         vpos[ncount++] = std::pair<int, int>( i, j );
00145       }
00146     }
00147 
00148     // choose a link between tow nodes according to a certain probability
00149     std::vector<double> vadjdist( n * n, dInfty );
00150     for( i = 0; i < n; ++i ) {
00151       vadjdist[i *n+i] = 0.0;
00152       for( j = i + 1; j < n; ++j ) {
00153         int d1 = vpos[i].first - vpos[j].first;
00154         int d2 = vpos[i].second - vpos[j].second;
00155         ddist = sqrt(( double )( d1 * d1 + d2 * d2 ) );
00156         dtmp = lamda * exp( -ddist / ( rou * dmaxdist ) );
00157         if( afl::rand::gen( 1.0 ) < dtmp ) {
00158           vadjdist[i *n+j] = ddist;
00159           vadjdist[j *n+i] = ddist;
00160           ++nlinks;
00161           //cout<<i<<", "<<j<<": "<<vadjdist[i*n+j]<<endl;
00162         }
00163       }// for j
00164     }// for i
00165 
00166     // find distances between all pairs
00167     std::vector<double> vdist( vadjdist );
00168     // vdist[j*n+i] is distance between (i,j)
00169     for( i = 0; i < n; ++i ) {
00170       for( j = 0; j < n; ++j )
00171         for( int k = 0; k < n; ++k )
00172           if( vdist[i*n+j] + vdist[k*n+i] < vdist[k*n+j] ) {
00173             vdist[k *n+j] = vdist[i*n+j] + vdist[k*n+i];
00174           }
00175     }
00176 
00177     // see if graph is connected
00178     bconnected = true;
00179     i = 0;
00180     while( i < n && bconnected ) {
00181       j = i + 1;
00182       while( j < n ) {
00183         //cout<<i<<", "<<j<<": "<<vdist[j*n+i]<<endl;
00184         if( vdist[j *n+i] >= dInfty ) {
00185           bconnected = false;
00186           break;
00187         }
00188         ++j;
00189       }
00190       ++i;
00191     }
00192 
00193     // get the graph from adj matrix, edge weight is the geometric distance
00194     if( bconnected ) {
00195       dgrf.reserve( n );
00196       for( i = 0; i < n; ++i ) {
00197         dgrf.insert_v( xyVertex<int>( vpos[i].first, vpos[i].second ) );
00198       }
00199       for( i = 0; i < n; ++i ) {
00200         for( j = i + 1; j < n; ++j ) {
00201           if( vadjdist[i*n+j] < dInfty ) {
00202             dgrf.insert_2e( xyVertex<int>( vpos[i].first, vpos[i].second ),
00203                     xyVertex<int>( vpos[j].first, vpos[j].second ),
00204                     vadjdist[j*n+i] );
00205           }
00206         }
00207       }
00208       return nlinks;
00209     }
00210 
00211     ntry--;
00212   }
00213 
00214   return -1;
